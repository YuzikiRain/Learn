基类

``` c++
// Animal.h
namespace MySpace
{
    class Animal
    {
        public:
        void Print();
        virtual Shout();
        static void Method();
        protected:
        string name;
        private:
    }
}
```

``` c++
// Animal.cpp
void MySpace::Animal::Print();
// 静态方法也是如此定义
void MySpace::Animal::Method();

// 或者
namespace MySpace
{
	void Animal::Print();
	void Animal::Method(); 
    // 这里不需要virtual
    void Animal::Shout(); 
}
```

派生类

``` c++
// Cat.h
namespace MySpace
{
    class Cat : public Animal
    {
        public:
        void Print();
        static void Method();
        protected:
        string name;
        private:
    }
}
```

``` c++
namespace MySpace
{
    // override了父类的实现
    void Cat::Shout()
    {
        // 显示调用父类Animal的实现，编译时就确定，效率更高
        Animal::Shout();
        // 通过虚拟机制，到运行时才知道调用的是父类Animal的Shout
        Shout();
    }
}
```

## 运行时的类型鉴定机制

``` c++
class A {}

class B : public A 
{
public:
    TestB(){}
}

void main()
{
    auto b = new B();
    A* a = b;
    if (typeid(*a) == typeid(B))
    {
        // 无条件转换
        B* b2 = static_cast<B*>(a);
    }
    // 先检查a所指向对象是否是B类型，如果是则转换操作发生，否则失败且返回0
    if (B* b3 = dynamic_cast<B*>(a))
    {
        b3->TestB();
    }
    delete b;
}
```
